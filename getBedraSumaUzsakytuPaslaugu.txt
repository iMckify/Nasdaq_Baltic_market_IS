SELECT sum(upa.kiekis) kiekis, sum(upa.kiekis*upa.kaina) suma 
FROM pavyzdys_paslaugos pa, pavyzdys_uzsakytos_paslaugos upa,
	pavyzdys_sutartys su
WHERE pa.id = upa.fk_paslauga and upa.fk_sutartis = su.nr
	AND su.sutarties_data >= IFNULL('2016-01-01', su.sutarties_data) 
    AND su.sutarties_data <= IFNULL('2017-01-01', su.sutarties_data)
